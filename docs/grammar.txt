Exp := Terms Exp_
Exp_ := '||' Terms Exp_ | Terms | LAMBDA

Terms := Comps Terms_
Terms_ := '&&' Comps Terms_ | Comps | LAMBDA

Comps :=    Factor    '!='  Factor | Factor   '=='  Factor 
            | Factor   '<'   Factor | Factor   '>'   Factor 
            | Factor   '='  Factor | Factor   '='  Factor 
            | Factor

Factor := '!'AriOp | AriOp

AriOp := AriOp2 AriOp_
AriOp_ := '+' AriOp2 AriOp_ | '-' AriOp2 AriOp_  |  AriOp2 | LAMBDA

AriOp2 := Parenthesis AriOp2_
AriOp2_ := '*' Parenthesis AriOp2_ | '/' Parenthesis AriOp2_  |  '%' Parenthesis AriOp2_ | Parenthesis | LAMBDA

Parenthesis := UnaryExp | '(' Exp ')' 

UnaryExp := '+'SimpleExp | '-'SimpleExp | SimpleExp 

SimpleExp := NumExp | AcessMemAddr | FuncCall 

NumExp := V_INT | V_REAL | V_BOOL | V_CHAR 

AcessMemAddr := Id | AcessMemAddr.Id | AcessMemAddr [Exp]
